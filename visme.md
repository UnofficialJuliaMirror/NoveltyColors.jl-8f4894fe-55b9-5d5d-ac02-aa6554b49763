      // parse a spec and create a visualization view
      function parse(divid, palette) {

        vg.parse.spec(spec, function(chart) { chart({el:"#" + divid}).update(); });
      }

      var visme;
      $.getJSON("/NoveltyColors.jl/javascripts/visme.json", function(json) {
          visme = json;
      })
